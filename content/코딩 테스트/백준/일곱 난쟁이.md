# 문제 내용

왕비를 피해 일곱 난쟁이들과 함께 평화롭게 생활하고 있던 백설공주에게 위기가 찾아왔다. 일과를 마치고 돌아온 난쟁이가 일곱 명이 아닌 아홉 명이었던 것이다.

아홉 명의 난쟁이는 모두 자신이 "백설 공주와 일곱 난쟁이"의 주인공이라고 주장했다. 뛰어난 수학적 직관력을 가지고 있던 백설공주는, 다행스럽게도 일곱 난쟁이의 키의 합이 100이 됨을 기억해 냈다.

아홉 난쟁이의 키가 주어졌을 때, 백설공주를 도와 일곱 난쟁이를 찾는 프로그램을 작성하시오.

## 입력

아홉 개의 줄에 걸쳐 난쟁이들의 키가 주어진다. 주어지는 키는 100을 넘지 않는 자연수이며, 아홉 난쟁이의 키는 모두 다르며, 가능한 정답이 여러 가지인 경우에는 아무거나 출력한다.

## 출력

일곱 난쟁이의 키를 오름차순으로 출력한다. 일곱 난쟁이를 찾을 수 없는 경우는 없다.

# 문제 분석

전부 다 더해가지고 100을 빼면 거짓말하고있는 두 사람의 키의 합이 나옴

그럼 그걸로 for문 두개 돌려서 완전탐색하면 될듯

# 작성한 코드

```cpp
#include<bits/stdc++.h>
using namespace std;
int main() {
    vector<int> height;
    for(int i = 0; i < 9; i++){
        int t;
        scanf("%d", &t);
        height.push_back(t);
    }
    int height_cha = accumulate(height.begin(), height.end(), 0) - 100;
    bool flag = false;
    for(int i = 0; i < height.size() - 1; i++){
        if(flag) break;
        for(int j = i+1; j < height.size(); j++){
            if(flag) break;
            if(height[i] + height[j] == height_cha){
                height.erase(height.begin() + i);
                height.erase(height.begin() + j - 1);
                flag = true;
            }
        }
    }
    sort(height.begin(), height.end());
    for(int i = 0; i < height.size(); i++){
        printf("%d\\n", height[i]);
    }
}
```

# 우수 코드 분석

#include <cstdio>

int s=-100, a[9], b[99];
int main() {
	int i, j;
	for(i=0; i<9; i++) {
		scanf("%d", &a[i]);
		s += a[i];
		b[a[i]] = 1;
	}
	for(i=0; i<9; i++) for(j=i+1; j<9; j++)
		if(a[i]+a[j] == s) b[a[i]] = b[a[j]] = 0, i = j = 9;
	for(i=0; i<99; i++) if(b[i]) printf("%d\\n", i);
	return 0;
}
